# Popular operators

SELECT id,
       email,
FROM databasename."Users"

LIMIT 10;
--  выгружает первые записи таблицы

LIMIT 10 OFFSET 4;
--  выгружает первые записи таблицы, начиная с 5

SELECT track_id, -- здесь указаны нужные поля
       album_id
       ...
FROM track -- таблица, из которой выгружают данные
WHERE track_id = 5; -- условие для среза данных

WHERE city = 'Saint-Petersburg'
  OR city = 'Kazan';

WHERE NOT city = 'Kazan';

WHERE (age > 43
  AND city = 'Saint-Petersburg')
  OR (city = 'Kazan'
  AND age < 43);

-- operator IN
SELECT total,
       customer_id
FROM invoice
WHERE billing_city IN ('Dublin',
                       'London',
                       'Paris',
                       'Boston',
                       'Berlin',
                       'Stuttgart');
-- instead of
SELECT total,
       customer_id
FROM invoice
WHERE billing_city = 'Dublin'
  OR billing_city = 'London'
  OR billing_city = 'Paris'
  OR billing_city = 'Boston'
  OR billing_city = 'Berlin'
  OR billing_city = 'Stuttgart';


-- BETWEEN
SELECT *
FROM invoice
WHERE customer_id BETWEEN 23 AND 48; 


SELECT last_name
FROM staff
WHERE birth_date BETWEEN '1970-01-01' AND '1970-12-31'; 

-- NULL
SELECT email,
       fax
FROM client
WHERE fax IS NULL -- выгрузить записи с NULL
LIMIT 5; 

SELECT email,
       fax
FROM client
WHERE fax IS NOT NULL -- выгрузить записи без NULL
LIMIT 5; 

-- operators LIKE and BETWEEN
WHERE quotes LIKE 'кот%'; -- Значения, которые начинаются на кот: 'кот просил еды' 'котелок не варит'

WHERE quotes LIKE 'ко%т'; -- Значения, которые начинаются на ко и заканчиваются на т, пример: 'кот играет' 'консультант'

# Data types

Группы типов

  • числовые типы
    • integer, int, int4 - целые числа
    • real, float4 - вещественные числа (могут быть не целые)
  • символические типы
    • character, char(n) - текст фиксированной длины, если для n=4, делается записть длиной в 2, недостающие символы будут заполнены пробелами; если не указывать n, значение по умолчанию = 1
    • character varying, varchar(n) - текст нефиксированной длины, n - лимит строки, необязательно указывать
  • логические типы
    • boolean: TRUE: 'true', 't', 'yes', 'y', 'on', '1' FALSE: 'false', 'f', 'no', 'n', 'off', '0'
  • типы для работы с датой и временем в PostgreSQL популярный формат 'YYYY-MM-DD'
    • timestamp, по умолчанию timestamp without time zone, пример: '2020-03-29 11:20:03'
    • timestamptz - с тайм-зоной, формат UTC, пример: '2020-03-29 11:20:03+02'
    • date - только дата, пример: 'October 19, 2004', '2004-Oct-19', '20041019'
    • time - только время от 00:00:00 до 24:00:00
  • разные специализированные типы для конкретных задач
-- CAST
SELECT CAST(id AS character)
-- меняет тип данных при выгрузке, например, если тип данных id - int, а нам нужно выгрузить как строку character
FROM table;

Хорошая база данных должна соответствовать нескольким критериям. Вот они:

  • БД целостная.
  • Данные можно использовать многократно.
  • Информация по запросам пользователей ищется быстро.
  • БД легко обновлять.
  • Данные надёжно защищены.
  • Нет избыточных данных.
